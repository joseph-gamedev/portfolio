    Intersection Algorithms

    Point Intersection

    Point-Point Intersection

    Explanation: Checking if two points in space are exactly the same.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y, z;
};

bool pointPointIntersection(const Point& p1, const Point& p2) {
    return p1.x == p2.x && p1.y == p2.y && p1.z == p2.z;
}

int main() {
    Point p1 = {1.0f, 2.0f, 3.0f};
    Point p2 = {1.0f, 2.0f, 3.0f};

    if (pointPointIntersection(p1, p2)) {
        std::cout << "Points are identical." << std::endl;
    } else {
        std::cout << "Points are different." << std::endl;
    }

    return 0;
}

    Point-Line Intersection

    Explanation: Determining if a point lies on a line defined by two other points.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y;
};

bool pointLineIntersection(const Point& p, const Point& lineStart, const Point& lineEnd) {
    float distStartToPoint = sqrt(pow(p.x - lineStart.x, 2) + pow(p.y - lineStart.y, 2));
    float distEndToPoint = sqrt(pow(p.x - lineEnd.x, 2) + pow(p.y - lineEnd.y, 2));
    float lineLength = sqrt(pow(lineEnd.x - lineStart.x, 2) + pow(lineEnd.y - lineStart.y, 2));

    // Check if the sum of distances from the point to the line endpoints equals the length of the line
    return std::abs(distStartToPoint + distEndToPoint - lineLength) < 0.001f; // Adjust epsilon for tolerance
}

int main() {
    Point p = {2.0f, 3.0f};
    Point lineStart = {1.0f, 1.0f};
    Point lineEnd = {5.0f, 5.0f};

    if (pointLineIntersection(p, lineStart, lineEnd)) {
        std::cout << "Point lies on the line." << std::endl;
    } else {
        std::cout << "Point does not lie on the line." << std::endl;
    }

    return 0;
}

    Point-Ray Intersection

    Explanation: Checking if a point intersects with an infinite ray emanating from a point with a given direction.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y;
};

bool pointRayIntersection(const Point& p, const Point& rayStart, const Point& rayDirection) {
    // Calculate vector from ray start to point
    float vecPX = p.x - rayStart.x;
    float vecPY = p.y - rayStart.y;

    // Calculate dot product to determine collinearity
    float dotProduct = vecPX * rayDirection.x + vecPY * rayDirection.y;

    // Check if point is on the ray (dot product non-negative and in the direction of the ray)
    return dotProduct >= 0.0f;
}

int main() {
    Point p = {2.0f, 3.0f};
    Point rayStart = {1.0f, 1.0f};
    Point rayDirection = {1.0f, 1.0f}; // Direction vector of the ray

    if (pointRayIntersection(p, rayStart, rayDirection)) {
        std::cout << "Point lies on the ray." << std::endl;
    } else {
        std::cout << "Point does not lie on the ray." << std::endl;
    }

    return 0;
}

    Point-Segment Intersection

    Explanation: Checking if a point lies on a line segment defined by two endpoints.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y;
};

bool pointSegmentIntersection(const Point& p, const Point& segStart, const Point& segEnd) {
    // Check if point lies on the line defined by the segment endpoints
    float crossProduct = (p.y - segStart.y) * (segEnd.x - segStart.x) - (p.x - segStart.x) * (segEnd.y - segStart.y);

    if (std::abs(crossProduct) > 0.001f) // Adjust epsilon for tolerance
        return false;

    float dotProduct = (p.x - segStart.x) * (segEnd.x - segStart.x) + (p.y - segStart.y) * (segEnd.y - segStart.y);
    if (dotProduct < 0)
        return false;

    float squaredLengthSE = (segEnd.x - segStart.x) * (segEnd.x - segStart.x) + (segEnd.y - segStart.y) * (segEnd.y - segStart.y);
    if (dotProduct > squaredLengthSE)
        return false;

    return true;
}

int main() {
    Point p = {2.0f, 3.0f};
    Point segStart = {1.0f, 1.0f};
    Point segEnd = {5.0f, 5.0f};

    if (pointSegmentIntersection(p, segStart, segEnd)) {
        std::cout << "Point lies on the segment." << std::endl;
    } else {
        std::cout << "Point does not lie on the segment." << std::endl;
    }

    return 0;
}

    Point-Plane Intersection

    Explanation: Determining if a point lies on a plane defined by a normal vector and a point on the plane.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y, z;
};

struct Plane {
    float a, b, c, d; // Equation of the plane: ax + by + cz + d = 0
};

bool pointPlaneIntersection(const Point& p, const Plane& plane) {
    float result = plane.a * p.x + plane.b * p.y + plane.c * p.z + plane.d;
    return std::abs(result) < 0.001f; // Adjust epsilon for tolerance
}

int main() {
    Point p = {1.0f, 2.0f, 3.0f};
    Plane plane = {1.0f, 2.0f, 3.0f, -10.0f}; // Example plane equation: x + 2y + 3z - 10 = 0

    if (pointPlaneIntersection(p, plane)) {
        std::cout << "Point lies on the plane." << std::endl;
    } else {
        std::cout << "Point does not lie on the plane." << std::endl;
    }

    return 0;
}

    Point-Triangle Intersection

    Explanation: Checking if a point lies inside or on the boundary of a triangle defined by three vertices.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y;
};

bool pointTriangleIntersection(const Point& p, const Point& t1, const Point& t2, const Point& t3) {
    // Calculate vectors
    float vec1X = t2.x - t1.x;
    float vec1Y = t2.y - t1.y;
    float vec2X = t3.x - t1.x;
    float vec2Y = t3.y - t1.y;
    float vecPX = p.x - t1.x;
    float vecPY = p.y - t1.y;

    // Calculate dot products
    float dot11 = vec1X * vec1X + vec1Y * vec1Y;
    float dot12 = vec1X * vec2X + vec1Y * vec2Y;
    float dot1P = vec1X * vecPX + vec1Y * vecPY;
    float dot22 = vec2X * vec2X + vec2Y * vec2Y;
    float dot2P = vec2X * vecPX + vec2Y * vecPY;

    // Calculate barycentric coordinates
    float invDenom = 1.0f / (dot11 * dot22 - dot12 * dot12);
    float u = (dot22 * dot1P - dot12 * dot2P) * invDenom;
    float v = (dot11 * dot2P - dot12 * dot1P) * invDenom;

    // Check if point is in triangle
    return (u >= 0.0f) && (v >= 0.0f) && (u + v <= 1.0f);
}

int main() {
    Point p = {2.0f, 3.0f};
    Point t1 = {1.0f, 1.0f};
    Point t2 = {5.0f, 1.0f};
    Point t3 = {3.0f, 5.0f};

    if (pointTriangleIntersection(p, t1, t2, t3)) {
        std::cout << "Point lies inside or on the triangle." << std::endl;


    } else {
        std::cout << "Point does not lie inside or on the triangle." << std::endl;
    }

    return 0;
}

    Point-Polygon Intersection

    Explanation: Determining if a point lies inside or on the boundary of a polygon defined by multiple vertices.

    C++ Example: 

    #include <iostream>
#include <vector>

struct Point {
    float x, y;
};

bool pointPolygonIntersection(const Point& p, const std::vector<Point>& polygon) {
    int numVertices = polygon.size();
    bool inside = false;

    for (int i = 0, j = numVertices - 1; i < numVertices; j = i++) {
        if (((polygon[i].y > p.y) != (polygon[j].y > p.y)) &&
            (p.x < (polygon[j].x - polygon[i].x) * (p.y - polygon[i].y) / (polygon[j].y - polygon[i].y) + polygon[i].x)) {
            inside = !inside;
        }
    }

    return inside;
}

int main() {
    Point p = {2.0f, 3.0f};
    std::vector<Point> polygon = {{1.0f, 1.0f}, {5.0f, 1.0f}, {3.0f, 5.0f}};

    if (pointPolygonIntersection(p, polygon)) {
        std::cout << "Point lies inside or on the polygon." << std::endl;
    } else {
        std::cout << "Point does not lie inside or on the polygon." << std::endl;
    }

    return 0;
}

    Point-Circle Intersection

    Explanation: Checking if a point lies inside, outside, or on the boundary of a circle defined by a center and radius.

    C++ Example: 

    #include <iostream>
#include <cmath>

struct Point {
    float x, y;
};

bool pointCircleIntersection(const Point& p, const Point& center, float radius) {
    float distanceSquared = pow(p.x - center.x, 2) + pow(p.y - center.y, 2);
    float radiusSquared = pow(radius, 2);

    // Check if the point is exactly on the circle's boundary
    if (std::abs(distanceSquared - radiusSquared) < 0.001f) // Adjust epsilon for tolerance
        return true;

    // Check if the point is inside the circle
    return distanceSquared < radiusSquared;
}

int main() {
    Point p = {2.0f, 3.0f};
    Point center = {1.0f, 1.0f};
    float radius = 5.0f;

    if (pointCircleIntersection(p, center, radius)) {
        std::cout << "Point lies inside or on the circle." << std::endl;
    } else {
        std::cout << "Point lies outside the circle." << std::endl;
    }

    return 0;
}

    Point-Box Intersection

    Vector3 Class

    We start by defining a simple Vector3 class to represent 3D points and vectors:

    class Vector3 {
public:
    float x, y, z;

    Vector3(float x = 0, float y = 0, float z = 0) : x(x), y(y), z(z) {}

    Vector3 operator-(const Vector3& v) const {
        return Vector3(x - v.x, y - v.y, z - v.z);
    }

    float dot(const Vector3& v) const {
        return x * v.x + y * v.y + z * v.z;
    }
};

    This class includes:

    • A constructor to initialize the vector
    • An overloaded subtraction operator for vector subtraction
    • A dot product method

    Axis-Aligned Bounding Box (AABB)

    Next, we define the AABB class:

    class AABB {
public:
    Vector3 min;
    Vector3 max;

    AABB(const Vector3& min, const Vector3& max) : min(min), max(max) {}

    bool intersects(const Vector3& point) const {
        return (point.x >= min.x && point.x <= max.x) &&
               (point.y >= min.y && point.y <= max.y) &&
               (point.z >= min.z && point.z <= max.z);
    }
};

    The AABB is defined by its minimum and maximum corners. The intersects method checks if a point is within or on the boundaries of the box.

    Oriented Bounding Box (OBB)

    The OBB class is more complex as it can be oriented in any direction:

    class OBB {
public:
    Vector3 center;
    Vector3 halfExtents;
    Vector3 axisX;
    Vector3 axisY;
    Vector3 axisZ;

    OBB(const Vector3& center, const Vector3& halfExtents, 
        const Vector3& axisX, const Vector3& axisY, const Vector3& axisZ)
        : center(center), halfExtents(halfExtents), axisX(axisX), axisY(axisY), axisZ(axisZ) {}

    bool intersects(const Vector3& point) const {
        Vector3 d = point - center;

        float projX = std::abs(d.dot(axisX));
        float projY = std::abs(d.dot(axisY));
        float projZ = std::abs(d.dot(axisZ));

        return projX <= halfExtents.x && projY <= halfExtents.y && projZ <= halfExtents.z;
    }
};

    The OBB is defined by:

    • Its center point
    • Half-extents (half the width, height, and depth)
    • Three axis vectors defining its orientation

    The intersects method:

    1. Calculates the vector from the center to the point
    2. Projects this vector onto each axis of the OBB
    3. Checks if these projections are within the half-extents

    Main Function

    Here's a simple main function to test our implementations:

    int main() {
    // Test AABB intersection
    AABB aabb(Vector3(-1, -1, -1), Vector3(1, 1, 1));
    Vector3 point1(0, 0, 0);
    Vector3 point2(2, 2, 2);

    std::cout << "AABB Intersection Test:" << std::endl;
    std::cout << "Point (0, 0, 0) intersects: " << (aabb.intersects(point1) ? "Yes" : "No") << std::endl;
    std::cout << "Point (2, 2, 2) intersects: " << (aabb.intersects(point2) ? "Yes" : "No") << std::endl;

    // Test OBB intersection
    OBB obb(Vector3(0, 0, 0), Vector3(1, 1, 1), 
            Vector3(1, 0, 0), Vector3(0, 1, 0), Vector3(0, 0, 1));
    Vector3 point3(0.5, 0.5, 0.5);
    Vector3 point4(1.5, 1.5, 1.5);

    std::cout << "\nOBB Intersection Test:" << std::endl;
    std::cout << "Point (0.5, 0.5, 0.5) intersects: " << (obb.intersects(point3) ? "Yes" : "No") << std::endl;
    std::cout << "Point (1.5, 1.5, 1.5) intersects: " << (obb.intersects(point4) ? "Yes" : "No") << std::endl;

    return 0;
}

    This main function creates instances of AABB and OBB, and tests point intersections with them.

    Conclusion

    This implementation provides efficient methods for testing point-box intersections for both AABB and OBB. The AABB intersection is simpler and faster, while the OBB intersection is more flexible but slightly more computationally expensive.

    Line Intersection

    Line-Line Intersection

    2D Case

    In 2D, two lines can be represented by equations: - Line 1: y = m1x + b1 - Line 2: y = m2x + b2

    The intersection point (x, y) is found by: - x = (b2 - b1) / (m1 - m2) - y = m1x + b1

    3D Case

    In 3D, lines are typically represented parametrically: - Line 1: P1 + t1(Q1 - P1) - Line 2: P2 + t2(Q2 - P2)

    Finding the intersection involves solving a system of equations. If the lines are skew (don't intersect), find the points of closest approach.

    // 2D Line-Line Intersection
std::optional<Vector2> intersect2DLines(const Vector2& p1, const Vector2& p2, 
                                        const Vector2& p3, const Vector2& p4) {
    float x1 = p1.x, y1 = p1.y, x2 = p2.x, y2 = p2.y;
    float x3 = p3.x, y3 = p3.y, x4 = p4.x, y4 = p4.y;

    float denom = (y4 - y3) * (x2 - x1) - (x4 - x3) * (y2 - y1);
    if (std::abs(denom) < 1e-6) return std::nullopt; // Lines are parallel

    float ua = ((x4 - x3) * (y1 - y3) - (y4 - y3) * (x1 - x3)) / denom;
    float ub = ((x2 - x1) * (y1 - y3) - (y2 - y1) * (x1 - x3)) / denom;

    if (ua < 0 || ua > 1 || ub < 0 || ub > 1) return std::nullopt; // Intersection point not on line segment

    float x = x1 + ua * (x2 - x1);
    float y = y1 + ua * (y2 - y1);

    return Vector2(x, y);
}

// 3D Line-Line Intersection (closest points)
std::pair<Vector3, Vector3> closestPointsBetweenLines(const Vector3& p1, const Vector3& p2, 
                                                      const Vector3& p3, const Vector3& p4) {
    Vector3 u = p2 - p1;
    Vector3 v = p4 - p3;
    Vector3 w = p1 - p3;

    float a = u.dot(u);
    float b = u.dot(v);
    float c = v.dot(v);
    float d = u.dot(w);
    float e = v.dot(w);
    float denom = a * c - b * b;

    float sc, tc;
    if (denom < 1e-6) {
        sc = 0;
        tc = (b > c ? d / b : e / c);
    } else {
        sc = (b * e - c * d) / denom;
        tc = (a * e - b * d) / denom;
    }

    return {p1 + u * sc, p3 + v * tc};
}

    Line-Plane Intersection

    A plane is defined by a point P and a normal vector n. A line is defined parametrically as L(t) = P0 + t(P1 - P0).

    The intersection point is found by:

    1. Calculate d = n (P - P0) / (n (P1 - P0))
    2. If d is between 0 and 1, the line intersects the plane at L(d)
    // Line-Plane Intersection
std::optional<Vector3> linePlaneIntersection(const Vector3& linePoint, const Vector3& lineDir,
                                             const Vector3& planePoint, const Vector3& planeNormal) {
    float denom = planeNormal.dot(lineDir);
    if (std::abs(denom) < 1e-6) return std::nullopt; // Line is parallel to plane

    float t = (planeNormal.dot(planePoint - linePoint)) / denom;
    return linePoint + lineDir * t;
}

    1. Complete Information: By returning a Vector3, we're providing the actual 3D point of intersection. This gives more information than just a scalar value, which would only tell us how far along the line the intersection occurs.

    2. Direct Usability: The returned Vector3 can be immediately used in further calculations or rendering without needing additional computation.

    3. Consistency with Other Intersection Functions: Many other intersection functions (like line-line or line-sphere) return points of intersection. Using Vector3 keeps the interface consistent across different types of intersections.

    4. Handling Special Cases: The std::optional wrapper allows us to handle cases where there is no intersection (like when the line is parallel to the plane) by returning std::nullopt.

    5. Avoiding Multiple Return Values: An alternative would be to return both a boolean (for intersection existence) and a float (for the intersection parameter). Returning a Vector3 in an std::optional combines both pieces of information into a single return value.

    6. Flexibility: While we could calculate t and return it, the caller would then need to know how to use t to find the intersection point. By returning the point directly, we encapsulate this logic in the function.

    If you specifically need the scalar parameter t, you could modify the function to return it, or add an output parameter for it:

    std::optional<Vector3> linePlaneIntersection(const Vector3& linePoint, const Vector3& lineDir,
                                             const Vector3& planePoint, const Vector3& planeNormal,
                                             float* outT = nullptr) {
    float denom = planeNormal.dot(lineDir);
    if (std::abs(denom) < 1e-6) return std::nullopt; // Line is parallel to plane

    float t = (planeNormal.dot(planePoint - linePoint)) / denom;
    if (outT) *outT = t;
    return linePoint + lineDir * t;
}

    This way, callers who need t can get it, while those who just want the intersection point don't need to handle an extra parameter.

    Line-Triangle Intersection

    This is often implemented using the MllerTrumbore algorithm:

    1. Represent the triangle by its vertices V0, V1, V2
    2. Use the parametric line equation
    3. Solve a system of equations to find barycentric coordinates
    4. If barycentric coordinates are all positive and sum to 1, there's an intersection
     // 13. Line-Triangle Intersection (MllerTrumbore algorithm)
std::optional<Vector3> lineTriangleIntersection(const Vector3& lineOrigin, const Vector3& lineDir,
                                                const Vector3& v0, const Vector3& v1, const Vector3& v2) {
    const float EPSILON = 0.0000001f;
    Vector3 edge1 = v1 - v0;
    Vector3 edge2 = v2 - v0;
    Vector3 h = lineDir.cross(edge2);
    float a = edge1.dot(h);

    if (a > -EPSILON && a < EPSILON) return std::nullopt; // Line is parallel to triangle

    float f = 1.0f / a;
    Vector3 s = lineOrigin - v0;
    float u = f * s.dot(h);

    if (u < 0.0f || u > 1.0f) return std::nullopt;

    Vector3 q = s.cross(edge1);
    float v = f * lineDir.dot(q);

    if (v < 0.0f || u + v > 1.0f) return std::nullopt;

    float t = f * edge2.dot(q);
    if (t > EPSILON) {
        return lineOrigin + lineDir * t;
    }

    return std::nullopt;
}

    Line-Polygon Intersection

    For a convex polygon:

    1. Intersect the line with each edge of the polygon
    2. If there are two intersection points within the polygon's bounds, the line intersects

    For a concave polygon:

    1. Use the even-odd rule or winding number algorithm
    2. Count intersections of the line with polygon edges
    // Line segment intersection
std::optional<Vector2> lineSegmentIntersection(const Vector2& p1, const Vector2& p2, 
                                               const Vector2& q1, const Vector2& q2) {
    Vector2 r = p2 - p1;
    Vector2 s = q2 - q1;
    float rxs = r.x * s.y - r.y * s.x;

    if (std::abs(rxs) < 1e-6) return std::nullopt; // Lines are parallel

    Vector2 qp = q1 - p1;
    float t = (qp.x * s.y - qp.y * s.x) / rxs;
    float u = (qp.x * r.y - qp.y * r.x) / rxs;

    if (t >= 0 && t <= 1 && u >= 0 && u <= 1) {
        return p1 + r * t;
    }

    return std::nullopt;
}

// Line-Polygon Intersection (handles both convex and concave polygons)
std::vector<Vector2> linePolygonIntersection(const Vector2& lineStart, const Vector2& lineEnd,
                                             const std::vector<Vector2>& polygonVertices) {
    std::vector<Vector2> intersections;
    int n = polygonVertices.size();

    // Check intersections with all edges
    for (int i = 0; i < n; i++) {
        const Vector2& v1 = polygonVertices[i];
        const Vector2& v2 = polygonVertices[(i + 1) % n];

        auto intersection = lineSegmentIntersection(lineStart, lineEnd, v1, v2);
        if (intersection) {
            intersections.push_back(*intersection);
        }
    }

    // Remove duplicates
    std::sort(intersections.begin(), intersections.end(), 
              [](const Vector2& a, const Vector2& b) { return a.x < b.x || (a.x == b.x && a.y < b.y); });
    intersections.erase(std::unique(intersections.begin(), intersections.end()), intersections.end());

    return intersections;
}

    Line-Circle Intersection

    Given a circle with center (h, k) and radius r, and a line y = mx + b:

    1. Substitute the line equation into the circle equation: (x - h) + (mx + b - k) = r
    2. Solve the resulting quadratic equation
    3. 0, 1, or 2 solutions indicate the number of intersection points
    // Line-Circle Intersection (2D)
std::vector<Vector2> lineCircleIntersection(const Vector2& linePoint, const Vector2& lineDir,
                                            const Vector2& circleCenter, float radius) {
    std::vector<Vector2> intersections;

    Vector2 d = linePoint - circleCenter;
    float a = lineDir.dot(lineDir);
    float b = 2 * d.dot(lineDir);
    float c = d.dot(d) - radius * radius;

    float discriminant = b * b - 4 * a * c;
    if (discriminant < 0) return intersections; // No intersection

    float t1 = (-b + std::sqrt(discriminant)) / (2 * a);
    float t2 = (-b - std::sqrt(discriminant)) / (2 * a);

    intersections.push_back(Vector2(linePoint.x + t1 * lineDir.x, linePoint.y + t1 * lineDir.y));
    if (discriminant > 0) { // Two intersections
        intersections.push_back(Vector2(linePoint.x + t2 * lineDir.x, linePoint.y + t2 * lineDir.y));
    }

    return intersections;
}

    Line-Sphere Intersection

    Similar to circle, but in 3D:

    1. Use the parametric line equation
    2. Substitute into the sphere equation: (x - cx) + (y - cy) + (z - cz) = r
    3. Solve the resulting quadratic equation in t
    // Line-Sphere Intersection
std::vector<Vector3> lineSphereIntersection(const Vector3& lineOrigin, const Vector3& lineDir,
                                            const Vector3& sphereCenter, float radius) {
    std::vector<Vector3> intersections;

    Vector3 oc = lineOrigin - sphereCenter;
    float a = lineDir.dot(lineDir);
    float b = 2.0f * oc.dot(lineDir);
    float c = oc.dot(oc) - radius * radius;
    float discriminant = b * b - 4 * a * c;

    if (discriminant < 0) return intersections; // No intersection

    float t1 = (-b + std::sqrt(discriminant)) / (2 * a);
    float t2 = (-b - std::sqrt(discriminant)) / (2 * a);

    intersections.push_back(lineOrigin + lineDir * t1);
    if (discriminant > 0) { // Two intersections
        intersections.push_back(lineOrigin + lineDir * t2);
    }

    return intersections;
}

    Line-Box Intersection

    For an Axis-Aligned Bounding Box (AABB):

    1. Use the slab method: intersect the line with the planes defined by the box faces
    2. Find the near and far intersection distances for each axis
    3. If the maximum of the near distances is less than the minimum of the far distances, there's an intersection

    For an Oriented Bounding Box (OBB):

    1. Transform the line into the OBB's local space
    2. Apply the AABB intersection method in this local space

    bool lineAABBIntersection(const Vector3& lineStart, const Vector3& lineEnd,
                          const Vector3& boxMin, const Vector3& boxMax) {
    Vector3 dir = lineEnd - lineStart;
    Vector3 invDir(1.0f / dir.x, 1.0f / dir.y, 1.0f / dir.z);

    float t1 = (boxMin.x - lineStart.x) * invDir.x;
    float t2 = (boxMax.x - lineStart.x) * invDir.x;
    float t3 = (boxMin.y - lineStart.y) * invDir.y;
    float t4 = (boxMax.y - lineStart.y) * invDir.y;
    float t5 = (boxMin.z - lineStart.z) * invDir.z;
    float t6 = (boxMax.z - lineStart.z) * invDir.z;

    float tmin = std::max(std::max(std::min(t1, t2), std::min(t3, t4)), std::min(t5, t6));
    float tmax = std::min(std::min(std::max(t1, t2), std::max(t3, t4)), std::max(t5, t6));

    return tmax >= tmin && tmax >= 0 && tmin <= 1;
}

    Ray Intersection

    Ray-Plane Intersection

    A plane in 3D space can be defined by the equation:
    [ ax + by + cz + d = 0 ] where \( (a, b, c) \) is the normal vector \( \mathbf{n}\) of the plane, and \( d\) is the distance from the origin.

    A ray is defined by its origin \(\mathbf{o}\) and direction \(\mathbf{d}\): $$ \mathbf{r}(t) = \mathbf{o} + t\mathbf{d} $$

    To find the intersection point of the ray with the plane:

    1. Substitute the ray equation into the plane equation: $$ a(\mathbf{o}_x + t\mathbf{d}_x) + b(\mathbf{o}_y + t\mathbf{d}_y) + c(\mathbf{o}_z + t\mathbf{d}_z) + d = 0 $$

    2. Simplify to solve for \( t \): $$ [ t = -\frac{\mathbf{n} \cdot \mathbf{o} + d}{\mathbf{n} \cdot \mathbf{d}} ] $$

    3. If \( \mathbf{n} \cdot \mathbf{d} \neq 0 \), there is a valid intersection at: $$ [ \mathbf{p} = \mathbf{o} + t\mathbf{d} ] $$

    #include <glm/glm.hpp>
#include <optional>

std::optional<float> RayPlaneIntersection(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& planeNormal, float planeD) {
    float denom = glm::dot(planeNormal, rayDir);
    if (denom != 0.0f) {
        float t = -(glm::dot(planeNormal, rayOrigin) + planeD) / denom;
        if (t >= 0) {
            return t;
        }
    }
    return std::nullopt;
}

    • Inputs:

      • rayOrigin: The starting point of the ray.
      • rayDir: The direction of the ray.
      • planeNormal: The normal vector of the plane.
      • planeD: The plane's D component in the plane equation \(ax + by + cz + d = 0\).

    • Steps:

      1. Compute the dot product of the plane normal and the ray direction. This value (denom) helps determine if the ray is parallel to the plane.
      2. If denom is not zero, calculate t using the plane equation.
      3. If t is non-negative, return t which is the distance along the ray direction where the intersection occurs.
      4. If denom is zero or t is negative, return std::nullopt, indicating no intersection or the intersection is behind the ray origin.

    Ray-Triangle Intersection

    A triangle in 3D space is defined by three vertices \( \mathbf{v}_0, \mathbf{v}_1, \mathbf{v}_2 \).

    To check if a ray intersects the triangle:

    1. Compute the plane normal: $$ \mathbf{n} = (\mathbf{v}_1 - \mathbf{v}_0) \times (\mathbf{v}_2 - \mathbf{v}_0) $$

    2. Find the intersection of the ray with the plane of the triangle (as above).

    3. Check if the intersection point \( \mathbf{p} \) lies inside the triangle using barycentric coordinates or edge tests:

      • Compute vectors: $$ \mathbf{v_0p} = \mathbf{p} - \mathbf{v}_0 $$ $$ \mathbf{v0v1} = \mathbf{v}_1 - \mathbf{v}_0 $$ $$ \mathbf{v0v2} = \mathbf{v}_2 - \mathbf{v}_0 $$

      • Solve for \( u \) and \( v \): $$ [ \mathbf{p} = (1 - u - v) \mathbf{v}_0 + u \mathbf{v}_1 + v \mathbf{v}_2 ] $$ Ensure \( 0 \leq u \leq 1 \), \( 0 \leq v \leq 1 \), and \( u + v \leq 1 \).

    #include <glm/glm.hpp>
#include <optional>

std::optional<float> RayTriangleIntersection(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& v0, const glm::vec3& v1, const glm::vec3& v2) {
    const float EPSILON = 0.0000001f;
    glm::vec3 edge1 = v1 - v0;
    glm::vec3 edge2 = v2 - v0;
    glm::vec3 h = glm::cross(rayDir, edge2);
    float a = glm::dot(edge1, h);
    if (a > -EPSILON && a < EPSILON) return std::nullopt;

    float f = 1.0f / a;
    glm::vec3 s = rayOrigin - v0;
    float u = f * glm::dot(s, h);
    if (u < 0.0f || u > 1.0f) return std::nullopt;

    glm::vec3 q = glm::cross(s, edge1);
    float v = f * glm::dot(rayDir, q);
    if (v < 0.0f || u + v > 1.0f) return std::nullopt;

    float t = f * glm::dot(edge2, q);
    if (t > EPSILON) return t;
    return std::nullopt;
}

    • Inputs:

      • rayOrigin: The starting point of the ray.
      • rayDir: The direction of the ray.
      • v0, v1, v2: The vertices of the triangle.

    • Steps:

      1. Compute two edges of the triangle.
      2. Compute the vector perpendicular to rayDir and edge2.
      3. Calculate a to check if the ray is parallel to the triangle.
      4. Compute f as the inverse of a.
      5. Calculate u to determine if the intersection point lies within the triangle bounds.
      6. Calculate q as the cross product of s and edge1.
      7. Calculate v to further ensure the point lies within the triangle.
      8. Calculate t, the intersection distance along the ray.
      9. Return t if valid; otherwise, return std::nullopt.

    Ray-Sphere Intersection

    A sphere is defined by its center \( \mathbf{c} \) and radius \( r \).

    To find the intersection of a ray with a sphere:

    1. Compute the vector from the ray origin to the sphere center: $$ \mathbf{l} = \mathbf{c} - \mathbf{o} $$

    2. Project \( \mathbf{l} \) onto the ray direction to find the closest approach: $$ t_{\text{ca}} = \mathbf{l} \cdot \mathbf{d} $$

    3. Find the squared distance from the sphere center to the ray: $$ d^2 = \mathbf{l} \cdot \mathbf{l} - t_{\text{ca}}^2 $$

    4. Check if the ray intersects the sphere: $$ r^2 \geq d^2 $$

    5. Calculate intersection distances: $$ t_h = \sqrt{r^2 - d^2} $$ $$ t_0 = t_{\text{ca}} - t_h $$ $$ t_1 = t_{\text{ca}} + t_h $$

    #include <glm/glm.hpp>
#include <optional>

std::optional<float> RaySphereIntersection(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& sphereCenter, float sphereRadius) {
    glm::vec3 L = sphereCenter - rayOrigin;
    float tca = glm::dot(L, rayDir);
    float d2 = glm::dot(L, L) - tca * tca;
    float radius2 = sphereRadius * sphereRadius;
    if (d2 > radius2) return std::nullopt;
    float thc = sqrt(radius2 - d2);
    float t0 = tca - thc;
    float t1 = tca + thc;
    if (t0 > t1) std::swap(t0, t1);
    if (t0 < 0) {
        t0 = t1;
        if (t0 < 0) return std::nullopt;
    }
    return t0;
}

    • Inputs:

      • rayOrigin: The starting point of the ray.
      • rayDir: The direction of the ray.
      • sphereCenter: The center of the sphere.
      • sphereRadius: The radius of the sphere.

    • Steps:

      1. Compute the vector from the ray origin to the sphere center.
      2. Project this vector onto the ray direction to find the closest approach.
      3. Calculate the squared distance from the sphere center to the ray.
      4. Check if this distance is within the sphere's radius.
      5. Compute the distances along the ray to the intersection points.
      6. Ensure the correct ordering of distances and check if they are valid.
      7. Return the smallest positive distance if valid; otherwise, return std::nullopt.

    Ray-Box Intersection (Ray-AABB Intersection)

    An Axis-Aligned Bounding Box (AABB) is defined by its minimum and maximum corners \( \mathbf{b}_{\min} \) and \( \mathbf{b}_{\max} \).

    To find the intersection of a ray with an AABB:

    1. Compute the intersection of the ray with the planes of the box: $$ t_{\min} = \frac{b_{\min} - \mathbf{o}}{\mathbf{d}} $$ $$ t_{\max} = \frac{b_{\max} - \mathbf{o}}{\mathbf{d}} $$

    2. Calculate the entry and exit distances: $$ t_{\text{entry}} = \max(t_{\min}) $$ $$ t_{\text{exit}} = \min(t_{\max}) $$

    3. Ensure there is a valid intersection: $$ t_{\text{entry}} \leq t_{\text{exit}} $$

      #include <glm/glm.hpp>
#include <algorithm>
#include <optional>

std::optional<float> RayAABBIntersection(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& boxMin, const glm::vec3& boxMax) {
    glm::vec3 invDir = 1.0f / rayDir;
    glm::vec3 t0 = (boxMin - rayOrigin) * invDir;
    glm::vec3 t1 = (boxMax - rayOrigin) * invDir;
    glm::vec3 tmin = glm::min(t0, t1);
    glm::vec3 tmax = glm::max(t0, t1);
    float tEnter = glm::compMax(tmin);
    float tExit = glm::compMin(tmax);
    if (tEnter > tExit || tExit < 0) return std::nullopt;
    return tEnter;
}

    4. Inputs:

      • rayOrigin: The starting point of the ray.
      • rayDir: The direction of the ray.
      • boxMin: The minimum corner of the AABB.
      • boxMax: The maximum corner of the AABB.

    5. Steps:

      1. Compute the inverse of the ray direction.
      2. Calculate the intersection times for the ray with the AABB planes.
      3. Determine the entry and exit times along each axis.
      4. Calculate the overall entry and exit times.
      5. Check if the ray enters before it exits and if the exit time is non-negative.
      6. Return the entry time if valid; otherwise, return std::nullopt.

    Ray-OBB (Oriented Bounding Box) Intersection

    An Oriented Bounding Box (OBB) is defined by its center \( \mathbf{c} \), local axes \( \mathbf{u}_i \), and half-extents \( e_i \).

    To find the intersection of a ray with an OBB:

    1. Transform the ray into the OBB's local space: $$ \mathbf{o}' = \mathbf{o} - \mathbf{c} $$ $$ \mathbf{o}' = \mathbf{M}^{-1} \mathbf{o}' $$ $$ \mathbf{d}' = \mathbf{M}^{-1} \mathbf{d} $$ where \( \mathbf{M} \) is the matrix of local axes \( \mathbf{u}_i \).

    2. Use the Ray-AABB intersection algorithm with the transformed ray and the AABB defined by the half-extents.

    These intersection tests are fundamental in many computer graphics and computational geometry applications, such as rendering, collision detection, and ray tracing.

    #include <glm/glm.hpp>
#include <glm/gtx/transform.hpp>
#include <optional>

std::optional<float> RayOBBIntersection(const glm::vec3& rayOrigin, const glm::vec3& rayDir, const glm::vec3& obbCenter, const glm::vec3& obbHalfExtents, const glm::mat3& obbOrientation) {
    glm::vec3 p = obbCenter - rayOrigin;
    glm::vec3 X = obbOrientation[0];
    glm::vec3 Y = obbOrientation[1];
    glm::vec3 Z = obbOrientation[2];

    glm::vec3 f(
        glm::dot(rayDir, X),
        glm::dot(rayDir, Y),
        glm::dot(rayDir, Z)
    );

    glm::vec3 e(
        glm::dot(p, X),
        glm::dot(p, Y),
        glm::dot(p, Z)
    );

    float t[6] = { 
        (e.x + obbHalfExtents.x) / f.x, 
        (e.x - obbHalfExtents.x) / f.x,
        (e.y + obbHalfExtents.y) / f.y, 
        (e.y - obbHalfExtents.y) / f.y,
        (e.z + obbHalfExtents.z) / f.z, 
        (e.z - obbHalfExtents.z) / f.z
    };

    float tmin = std::max({ std::min(t[0], t[1]), std::min(t[2], t[3]), std::min(t[4], t[5]) });
    float tmax = std::min({ std::max(t[0], t[1]), std::max(t[2],

 t[3]), std::max(t[4], t[5]) });

    if (tmax < 0) return std::nullopt;
    if (tmin > tmax) return std::nullopt;
    return tmin;
}

    • Inputs:

      • rayOrigin: The starting point of the ray.
      • rayDir: The direction of the ray.
      • obbCenter: The center of the OBB.
      • obbHalfExtents: Half the dimensions of the OBB.
      • obbOrientation: The orientation matrix of the OBB.

    • Steps:

      1. Compute the vector from the ray origin to the OBB center.
      2. Extract the local axes from the OBB orientation matrix.
      3. Project the ray direction onto the OBB axes.
      4. Project the vector from the ray origin to the OBB center onto the OBB axes.
      5. Calculate the intersection times for the ray with the OBB planes.
      6. Determine the entry and exit times along each axis.
      7. Calculate the overall entry and exit times.
      8. Check if the ray enters before it exits and if the exit time is non-negative.
      9. Return the entry time if valid; otherwise, return std::nullopt.

    Segment-Segment Intersection (2D)

    Explanation:

    To check if two line segments intersect in 2D, we can use the orientation method to determine the relative positions of the endpoints.

    C++ Implementation:

    #include <glm/glm.hpp>

// Function to find the orientation of the ordered triplet (p, q, r).
int orientation(const glm::vec2& p, const glm::vec2& q, const glm::vec2& r) {
    float val = (q.y - p.y) * (r.x - q.x) - (q.x - p.x) * (r.y - q.y);
    if (val == 0) return 0;  // collinear
    return (val > 0) ? 1 : 2;  // clock or counterclock wise
}

// Function to check if point q lies on segment pr.
bool onSegment(const glm::vec2& p, const glm::vec2& q, const glm::vec2& r) {
    return q.x <= std::max(p.x, r.x) && q.x >= std::min(p.x, r.x) &&
           q.y <= std::max(p.y, r.y) && q.y >= std::min(p.y, r.y);
}

bool segmentSegmentIntersection(const glm::vec2& p1, const glm::vec2& q1, const glm::vec2& p2, const glm::vec2& q2) {
    // Find the four orientations needed for general and special cases
    int o1 = orientation(p1, q1, p2);
    int o2 = orientation(p1, q1, q2);
    int o3 = orientation(p2, q2, p1);
    int o4 = orientation(p2, q2, q1);

    // General case
    if (o1 != o2 && o3 != o4) return true;

    // Special cases
    if (o1 == 0 && onSegment(p1, p2, q1)) return true;
    if (o2 == 0 && onSegment(p1, q2, q1)) return true;
    if (o3 == 0 && onSegment(p2, p1, q2)) return true;
    if (o4 == 0 && onSegment(p2, q1, q2)) return true;

    return false;
}

    Segment-Plane Intersection

    Explanation:

    To check if a segment intersects a plane, we need to find the point where the segment intersects the plane, if it exists. This can be done by solving the parametric line equation with the plane equation.

    C++ Implementation:

    #include <glm/glm.hpp>

// Plane is defined by a point and a normal vector
struct Plane {
    glm::vec3 point;
    glm::vec3 normal;
};

bool segmentPlaneIntersection(const glm::vec3& p1, const glm::vec3& p2, const Plane& plane, glm::vec3& intersection) {
    glm::vec3 segmentDir = p2 - p1;
    float denom = glm::dot(plane.normal, segmentDir);

    if (fabs(denom) > std::numeric_limits<float>::epsilon()) { // Segment is not parallel to the plane
        float t = glm::dot(plane.normal, plane.point - p1) / denom;
        if (t >= 0.0f && t <= 1.0f) { // Intersection point is within the segment
            intersection = p1 + t * segmentDir;
            return true;
        }
    }

    return false;
}

    Segment-Triangle Intersection

    Explanation:

    To check if a segment intersects a triangle, we can use the MllerTrumbore intersection algorithm.

    C++ Implementation:

    #include <glm/glm.hpp>

bool segmentTriangleIntersection(const glm::vec3& p1, const glm::vec3& p2, const glm::vec3& v0, const glm::vec3& v1, const glm::vec3& v2, glm::vec3& intersection) {

    glm::vec3 s = p1 - v0;
    float u = f * glm::dot(s, h);
    if (u < 0.0f || u > 1.0f) return false;

    glm::vec3 q = glm::cross(s, edge1);
    float v = f * glm::dot(segmentDir, q);
    if (v < 0.0f || u + v > 1.0f) return false;

    float t = f * glm::dot(edge2, q);
    if (t >= 0.0f && t <= 1.0f) {
        intersection = p1 + t * segmentDir;
        return true;
    }

    return false;
}

    Segment-Circle Intersection (2D)

    Explanation:

    To check if a segment intersects a circle, we can solve the quadratic equation derived from the parametric equation of the segment and the equation of the circle.

    C++ Implementation:

    #include <glm/glm.hpp>

bool segmentCircleIntersection(const glm::vec2& p1, const glm::vec2& p2, const glm::vec2& center, float radius, glm::vec2& intersection1, glm::vec2& intersection2) {
    glm::vec2 d = p2 - p1;
    glm::vec2 f = p1 - center;
    float a = glm::dot(d, d);
    float b = 2.0f * glm::dot(f, d);
    float c = glm::dot(f, f) - radius * radius;

    float discriminant = b * b - 4 * a * c;
    if (discriminant < 0) return false;

    discriminant = sqrt(discriminant);
    float t1 = (-b - discriminant) / (2 * a);
    float t2 = (-b + discriminant) / (2 * a);

    if (t1 >= 0 && t1 <= 1) {
        intersection1 = p1 + t1 * d;
    }

    if (t2 >= 0 && t2 <= 1) {
        intersection2 = p1 + t2 * d;
    }

    return t1 >= 0 && t1 <= 1 || t2 >= 0 && t2 <= 1;
}

    Segment-Box Intersection (AABB)

    Explanation:

    To check if a segment intersects an axis-aligned bounding box (AABB), we can use the slab method, which checks the segment against the

    planes of the AABB.

    C++ Implementation:

    #include <glm/glm.hpp>

bool segmentAABBIntersection(const glm::vec3& p1, const glm::vec3& p2, const glm::vec3& min, const glm::vec3& max, glm::vec3& intersection) {
    glm::vec3 d = p2 - p1;
    float tmin = 0.0f;
    float tmax = 1.0f;

    for (int i = 0; i < 3; ++i) {
        if (fabs(d[i]) < std::numeric_limits<float>::epsilon()) {
            if (p1[i] < min[i] || p1[i] > max[i]) return false;
        } else {
            float ood = 1.0f / d[i];
            float t1 = (min[i] - p1[i]) * ood;
            float t2 = (max[i] - p1[i]) * ood;
            if (t1 > t2) std::swap(t1, t2);
            tmin = std::max(tmin, t1);
            tmax = std::min(tmax, t2);
            if (tmin > tmax) return false;
        }
    }

    intersection = p1 + tmin * d;
    return true;
}

    Box Intersection

    AABB-AABB Intersection

    Explanation:

    AABB intersection is straightforward because both boxes are axis-aligned. The idea is to check if there is any overlap along each axis (x, y, and z). If there is overlap on all three axes, the boxes intersect.

    C++ Implementation:

    #include <glm/glm.hpp>

bool AABBIntersection(const glm::vec3& min1, const glm::vec3& max1, const glm::vec3& min2, const glm::vec3& max2) {
    // Check for overlap on the x-axis
    if (max1.x < min2.x || min1.x > max2.x) return false;
    // Check for overlap on the y-axis
    if (max1.y < min2.y || min1.y > max2.y) return false;
    // Check for overlap on the z-axis
    if (max1.z < min2.z || min1.z > max2.z) return false;

    // If there is overlap on all axes, the boxes intersect
    return true;
}

    OBB-OBB Intersection

    Explanation:

    OBB intersection is more complex because the boxes can be oriented in any direction. The Separating Axis Theorem (SAT) is typically used for this purpose. SAT states that two convex shapes do not overlap if there exists a plane (axis) that separates them. For OBBs, we need to test 15 potential separating axes: 1. The normals of each box's faces (3 axes from each box, total 6 axes). 2. The cross products of the edges of the two boxes (9 axes).

    C++ Implementation:

    #include <glm/glm.hpp>
#include <glm/gtx/transform.hpp>
#include <array>

bool OBBIntersection(const glm::vec3& center1, const glm::vec3& halfExtents1, const glm::mat3& orientation1,
                     const glm::vec3& center2, const glm::vec3& halfExtents2, const glm::mat3& orientation2) {
    // Compute rotation matrix expressing box2 in box1's coordinate frame
    glm::mat3 R;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            R[i][j] = glm::dot(orientation1[i], orientation2[j]);

    // Compute translation vector
    glm::vec3 t = center2 - center1;
    // Bring translation into box1's coordinate frame
    t = glm::vec3(glm::dot(t, orientation1[0]), glm::dot(t, orientation1[1]), glm::dot(t, orientation1[2]));

    // Compute common subexpressions
    glm::mat3 AbsR;
    bool existsParallelPair = false;
    const float EPSILON = 1e-6f;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++) {
            AbsR[i][j] = fabs(R[i][j]) + EPSILON;
            if (AbsR[i][j] >= 1.0f) existsParallelPair = true;
        }

    // Test axes L = A0, L = A1, L = A2
    for (int i = 0; i < 3; i++) {
        float ra = halfExtents1[i];
        float rb = halfExtents2[0] * AbsR[i][0] + halfExtents2[1] * AbsR[i][1] + halfExtents2[2] * AbsR[i][2];
        if (fabs(t[i]) > ra + rb) return false;
    }

    // Test axes L = B0, L = B1, L = B2
    for (int i = 0; i < 3; i++) {
        float ra = halfExtents1[0] * AbsR[0][i] + halfExtents1[1] * AbsR[1][i] + halfExtents1[2] * AbsR[2][i];
        float rb = halfExtents2[i];
        if (fabs(t[0] * R[0][i] + t[1] * R[1][i] + t[2] * R[2][i]) > ra + rb) return false;
    }

    // Test axes L = A0 x B0, L = A0 x B1, L = A0 x B2
    // Test axes L = A1 x B0, L = A1 x B1, L = A1 x B2
    // Test axes L = A2 x B0, L = A2 x B1, L = A2 x B2
    std::array<glm::vec3, 3> axes1 = {orientation1[0], orientation1[1], orientation1[2]};
    std::array<glm::vec3, 3> axes2 = {orientation2[0], orientation2[1], orientation2[2]};

    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            glm::vec3 axis = glm::cross(axes1[i], axes2[j]);
            float ra = halfExtents1[(i+1)%3] * AbsR[(i+2)%3][j] + halfExtents1[(i+2)%3] * AbsR[(i+1)%3][j];
            float rb = halfExtents2[(j+1)%3] * AbsR[i][(j+2)%3] + halfExtents2[(j+2)%3] * AbsR[i][(j+1)%3];
            if (fabs(glm::dot(t, axis)) > ra + rb) return false;
        }
    }

    return true;
}

    Detailed Explanation

    AABB-AABB Intersection

    1. Checking Overlap:
      • For each axis (x, y, z), check if the maximum extent of one box is less than the minimum extent of the other box. If this condition is true for any axis, there is no intersection.
      • If overlap exists on all three axes, the boxes intersect.

    OBB-OBB Intersection

    1. Separating Axis Theorem (SAT):

      • SAT states that two convex shapes do not overlap if there exists an axis along which the projection of the shapes does not overlap.
      • For OBBs, we need to test 15 potential separating axes.

    2. Compute Rotation Matrix:

      • Compute a 3x3 matrix expressing the orientation of box2 in box1's coordinate frame using dot products of the orientation axes.

    3. Translation Vector:

      • Calculate the translation vector from the center of box1 to box2, and express it in box1's coordinate frame.

    4. Common Subexpressions:

      • Precompute the absolute values of the rotation matrix elements to avoid redundant calculations and add a small epsilon to handle numerical errors.

    5. Test Face Normals:

      • Test the axes aligned with the face normals of both boxes (6 tests).

    6. Test Cross Products of Edges:

      • Test the cross products of the edges of the two boxes (9 tests).

    7. Projection and Overlap:

      • For each axis, project the half-extents of both boxes onto the axis.
      • Compute the distance of the projected centers along the axis.
      • If the distance is greater than the sum of the projected half-extents for any axis, the boxes do not intersect.

    8. Intersection Result:

      • If no separating axis is found, the boxes intersect. Return true.

    Convex Shape Intersection

    GJK (Gilbert-Johnson-Keerthi) Algorithm

    Explanation:

    The GJK algorithm is used to determine whether two convex shapes intersect. It does this by iteratively building a simplex (a triangle in 2D or a tetrahedron in 3D) that gets closer and closer to the Minkowski difference of the shapes, ultimately checking if the origin is within this simplex.

    C++ Implementation:

    Below is a simplified version of the GJK algorithm for 3D shapes.

    #include <glm/glm.hpp>
#include <vector>
#include <algorithm>
#include <limits>

struct Simplex {
    std::vector<glm::vec3> points;

    void add(const glm::vec3& point) {
        points.push_back(point);
    }

    void removeLast() {
        points.pop_back();
    }

    glm::vec3& operator[](size_t i) {
        return points[i];
    }

    size_t size() const {
        return points.size();
    }
};

glm::vec3 support(const std::vector<glm::vec3>& shape1, const std::vector<glm::vec3>& shape2, const glm::vec3& dir) {
    auto furthestPointInDirection = [](const std::vector<glm::vec3>& shape, const glm::vec3& dir) {
        float maxDot = -std::numeric_limits<float>::max();
        glm::vec3 furthestPoint;
        for (const auto& point : shape) {
            float dotProduct = glm::dot(point, dir);
            if (dotProduct > maxDot) {
                maxDot = dotProduct;
                furthestPoint = point;
            }
        }
        return furthestPoint;
    };

    glm::vec3 p1 = furthestPointInDirection(shape1, dir);
    glm::vec3 p2 = furthestPointInDirection(shape2, -dir);
    return p1 - p2;
}

bool GJK(const std::vector<glm::vec3>& shape1, const std::vector<glm::vec3>& shape2) {
    glm::vec3 initialDir(1, 0, 0);
    Simplex simplex;
    simplex.add(support(shape1, shape2, initialDir));

    glm::vec3 dir = -simplex[0];

    while (true) {
        simplex.add(support(shape1, shape2, dir));
        if (glm::dot(simplex[simplex.size() - 1], dir) <= 0) {
            return false;
        } else {
            if (simplex.size() == 2) {
                glm::vec3 a = simplex[1];
                glm::vec3 b = simplex[0];
                glm::vec3 ab = b - a;
                glm::vec3 ao = -a;
                dir = glm::cross(glm::cross(ab, ao), ab);
            } else if (simplex.size() == 3) {
                glm::vec3 a = simplex[2];
                glm::vec3 b = simplex[1];
                glm::vec3 c = simplex[0];
                glm::vec3 ab = b - a;
                glm::vec3 ac = c - a;
                glm::vec3 ao = -a;
                glm::vec3 abc = glm::cross(ab, ac);

                if (glm::dot(glm::cross(abc, ac), ao) > 0) {
                    simplex.removeLast();
                    dir = glm::cross(glm::cross(ac, ao), ac);
                } else if (glm::dot(glm::cross(ab, abc), ao) > 0) {
                    simplex.removeLast();
                    dir = glm::cross(glm::cross(ab, ao), ab);
                } else {
                    return true;
                }
            } else if (simplex.size() == 4) {
                glm::vec3 a = simplex[3];
                glm::vec3 b = simplex[2];
                glm::vec3 c = simplex[1];
                glm::vec3 d = simplex[0];
                glm::vec3 ab = b - a;
                glm::vec3 ac = c - a;
                glm::vec3 ad = d - a;
                glm::vec3 ao = -a;
                glm::vec3 abc = glm::cross(ab, ac);
                glm::vec3 acd = glm::cross(ac, ad);
                glm::vec3 adb = glm::cross(ad, ab);

                if (glm::dot(abc, ao) > 0) {
                    simplex.removeLast();
                    dir = abc;
                } else if (glm::dot(acd, ao) > 0) {
                    simplex.removeLast();
                    dir = acd;
                } else if (glm::dot(adb, ao) > 0) {
                    simplex.removeLast();
                    dir = adb;
                } else {
                    return true;
                }
            }
        }
    }
}

    SAT (Separating Axis Theorem)

    Explanation:

    The SAT is a method for determining whether two convex shapes are intersecting. It states that if there is a plane (axis) along which the projections of the two shapes do not overlap, then the shapes do not intersect. For convex polyhedra, the potential separating axes include the normals of all faces of both shapes and the cross products of all pairs of edges.

    C++ Implementation:

    Below is an implementation of SAT for convex polyhedra. This implementation assumes the polyhedra are represented as a set of vertices and a set of faces (each face is a set of vertex indices).

    #include <glm/glm.hpp>
#include <vector>
#include <algorithm>
#include <limits>

struct Face {
    std::vector<int> indices;
};

bool OverlapOnAxis(const std::vector<glm::vec3>& shape1, const std::vector<glm::vec3>& shape2, const glm::vec3& axis) {
    auto project = [&](const std::vector<glm::vec3>& shape, const glm::vec3& axis) {
        float minProj = std::numeric_limits<float>::max();
        float maxProj = -std::numeric_limits<float>::max();
        for (const auto& vertex : shape) {
            float projection = glm::dot(vertex, axis);
            if (projection < minProj) minProj = projection;
            if (projection > maxProj) maxProj = projection;
        }
        return std::make_pair(minProj, maxProj);
    };

    auto [min1, max1] = project(shape1, axis);
    auto [min2, max2] = project(shape2, axis);

    return !(max1 < min2 || max2 < min1);
}

bool SAT(const std::vector<glm::vec3>& shape1, const std::vector<glm::vec3>& shape2, const std::vector<Face>& faces1, const std::vector<Face>& faces2) {
    std::vector<glm::vec3> axes;

    // Normals of the faces of shape1
    for (const auto& face : faces1) {
        glm::vec3 normal = glm::normalize(glm::cross(shape1[face.indices[1]] - shape1[face.indices[0]], shape1[face.indices[2]] - shape1[face.indices[0]]));
        axes.push_back(normal);
    }

    // Normals of the faces of shape2
    for (const auto& face : faces2) {
        glm::vec3 normal = glm::normalize(glm::cross(shape2[face.indices[1]] - shape2[face.indices[0]], shape2[face.indices[2]] - shape2[face.indices[0]]));
        axes.push_back(normal);
    }

    // Cross products of edges
    auto addCrossProducts = [&](const std::vector<glm::vec3>& shape1, const std::vector<Face>& faces1, const std::vector<glm::vec3>& shape2, const std::vector<Face>& faces2) {
        for (const auto& face1 : faces1) {
            for (size_t i = 0; i < face1.indices.size(); ++i) {
                glm::vec3 edge1 = shape1[face1.indices[(i + 1) % face1.indices.size()]] - shape1[face1.indices[i]];
                for (const auto& face2 : faces2) {
                    for (size_t j = 0; j < face2.indices.size(); ++j) {
                        glm::vec3 edge2 = shape2[face2.indices[(j + 1) % face2.indices.size()]] - shape2[face2.indices[j]];
                        glm::vec3 axis = glm::normalize(glm::cross(edge1, edge2));
                        if (glm::length(axis) > std::numeric_limits<float>::epsilon()) {
                            axes.push_back(axis);
                        }
                    }
                }
            }
        }
    };

    addCrossProducts(shape1, faces1, shape2, faces2);

    for (const auto& axis : axes) {
        if (!OverlapOnAxis(shape1, shape2, axis)) {
            return false;
        }
    }

    return true;
}

    Detailed Explanation

    GJK Algorithm

    1. Support Function:

      • The support function computes the farthest point in a given direction for the Minkowski difference of the two shapes.

    2. Simplex Management:

      • The simplex is a set of

    points that the algorithm iteratively builds. - It starts with an initial direction and adds the support point in that direction.

    1. Direction Calculation:

      • The algorithm computes a new search direction based on the simplex and the origin's relative position.

    2. Termination Conditions:

      • If the new support point does not move past the origin, the shapes do not intersect.
      • If a valid simplex containing the origin is formed, the shapes intersect.

    SAT Algorithm

    1. Separating Axis Calculation:

      • For each face of the shapes, compute the face normal and add it as a potential separating axis.
      • For each pair of edges from the two shapes, compute the cross product and add it as a potential separating axis.

    2. Projection and Overlap Test:

      • Project both shapes onto each axis.
      • Check for overlap on each axis. If any axis exists where projections do not overlap, the shapes do not intersect.

    3. Intersection Result:

      • If no separating axis is found, the shapes intersect. Return true.
